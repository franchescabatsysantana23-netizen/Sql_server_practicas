Ejemplo de la Funcion Contains

select * from person.Address A
where contains (A.AddressLine1,'"first avenue"')

uso de constains con los operadores and/and not/or

select * from person.Address A
where contains (A.AddressLine1,'"avenue and port and not 34"')

select * from person.Address A
where contains (A.AddressLine1,'"avenue or port"')


select * from person.Address A
where contains (A.AddressLine1,'"(36 and avenue) or port"')

uso de la function near--> permite buscar entre dos palabras en una busqueda de tipo sql.
ejemplo
busca los resultados que contenga la palabra san near

select * from person.Address A
where contains(A.AddressLine1,'"5415 San"NEAR dr')

select * from person.Address A
 where contains(A.AddressLine1,'Near((21,centrale),5)')

--uso del predicado formsof inflectional(Sustantivos)

select * from person.address A
where contains (A.AddressLine1,'Formsof(inflectional,"Avenues Street")') ---> sirve para retornar todas las direcciones que tengan coincidencias.

--uso del predicado formsof inflectional (Verbos)

select * from person.address A
where contains (A.AddressLine1,'Formsof(inflectional,"hagamos")',language 3082)

El codigo 3082 corresponde a España. 

--Funcion para la obtención de tiempos verbales.

select display_term
from sys.dm_fts_parser('formsof(inflectional,saltar)',3082,0,0)
